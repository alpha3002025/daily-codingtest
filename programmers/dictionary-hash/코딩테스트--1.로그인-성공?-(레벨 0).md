def solution(id_pw, db):
    id_matched = False
    
    for row in db:
        if id_pw[0] == row[0] and id_pw[1] == row[1]:
            return "login"
        else:
            if id_pw[0] == row[0]:
                id_matched = True

    return "wrong pw" if id_matched else "fail"
