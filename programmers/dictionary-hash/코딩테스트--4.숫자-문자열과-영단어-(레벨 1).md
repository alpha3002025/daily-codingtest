# python (try1)
```python
def solution(s):    
    table = {
        "zero": 0,
        "one": 1,
        "two": 2,
        "three": 3,
        "four": 4,
        "five": 5,
        "six": 6,
        "seven": 7,
        "eight": 8,
        "nine": 9,
        
    }
    
    for word,value in table.items():
        if word in s:
            s = s.replace(word, str(value))
    
    return int(s)
```



# python (예전에 풀었던)
```python
def solution(s):
    word_to = {
        'zero': '0',
        'one': '1',
        'two': '2',
        'three': '3',
        'four': '4',
        'five': '5',
        'six': '6',
        'seven': '7',
        'eight': '8',
        'nine': '9'
        
    }
    
    for key,value in word_to.items():
        s = s.replace(key, word_to[key])
    
    answer = int(s)
    return answer
```
