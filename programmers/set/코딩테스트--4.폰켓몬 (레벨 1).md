```python
from collections import Counter

def solution(nums):
    counter = Counter(nums)
    res = sorted(counter.items(), key = lambda x: x[1], reverse=False)
    
    answer = 0
    
    n = len(nums)//2
    for i in range(n):
        if i < len(res):
            answer += 1
    
    return answer
```
