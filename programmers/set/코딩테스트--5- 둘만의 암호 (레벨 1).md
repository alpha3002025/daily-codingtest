```python
def solution(s, skip, index):
    skip_set = set()
    
    for c in skip:
        skip_set.add(ord(c))
    
    answer = ''
    for c in s:
        count = 0
        curr = ord(c)
        
        while count < index:
            if curr in skip_set:
                curr+=1
                continue
            curr+=1
            if ord('z')+1== curr:
                curr = ord('a')
            count+=1
            
        answer+=chr(curr)
    
    return answer
```


# python (2) (중간에 실수한 오답)
```python
def solution(s, skip, index):
    print(f"a = {ord('a')}, z = {ord('z')}")
    
    table = [0] * (26)
    
    for c in skip:
        table[ord(c)-ord('a')] = -1
    
    answer = ''
    for c in s:
        curr = ord(c) - ord('a')
        
        i = 0
        while i < index:
            curr = (curr + i - ord('a'))%26 ## (1), (2)
            if table[curr] == -1:
                continue
            i+=1
        
        answer += chr(curr) ## (3)
        
    return answer
```

# python (2) 를 고친 정답
- (1) i 만큼을 더하는 실수를 했다. 1을 더해야 했다. i는 i 가 0 \~ index 로 변하기에  +1, +2, +3 ... 이 되어서 이상하게 더해지게 된다. 조심
- (2) - ord('a') 를 한번 더했다.
- (3) `answer += chr(curr)` 이 아니라 `answer += chr(curr + ord('a'))` 을 해야 한다.
```python
def solution(s, skip, index):
    print(f"a = {ord('a')}, z = {ord('z')}")
    
    table = [0] * (26)
    
    for c in skip:
        table[ord(c)-ord('a')] = -1
    
    answer = ''
    for c in s:
        curr = ord(c) - ord('a')
        
        i = 0
        while i < index:
            curr = (curr + 1) % 26
            if table[curr] == -1:
                continue
            i+=1
        
        answer += chr(curr + ord('a'))
        
    return answer
```

```python
def solution(s, skip, index):
    table = [0] * 26
    
    for c in skip:
        table[ord(c) - ord('a')] = -1
    
    answer = ''
    for c in s:
        curr = ord(c) - ord('a')
        
        moved = 0  # 실제로 이동한 횟수
        while moved < index:
            curr = (curr + 1) % 26  # 다음 문자로 이동
            if table[curr] != -1:  # skip에 없는 문자만 카운트
                moved += 1
        
        answer += chr(curr + ord('a'))  # 문자로 변환
        
    return answer
```
