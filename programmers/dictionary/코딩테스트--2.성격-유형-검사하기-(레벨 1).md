2번 지표	콘형(C), 프로도형(F)
3번 지표	제이지형(J), 무지형(M)
4번 지표	어피치형(A), 네오형(N)
"""
def solution(survey, choices):
    table = ['RT', 'CF', 'JM', 'AN']
    counter = Counter()
    
    for s, c in zip(survey, choices):
        if 1 <= c <=3:
            counter[s[0]]+= c
        else:
            counter[s[1]]+= c%4

    print(f"{counter}")
    answer = ''
    answer += ('R' if counter['T'] <= counter['R'] else 'T')

    answer += ('J' if counter['M'] <= counter['J'] else 'M')
    answer += ('A' if counter['N'] <= counter['A'] else 'N')
    return answer

```python
from collections import Counter

def score(counter):
    ans = []
    
    ans.append('R' if counter['R'] >= counter['T'] else 'T')
    ans.append('C' if counter['C'] >= counter['F'] else 'F')
    ans.append('J' if counter['J'] >= counter['M'] else 'M')
    ans.append('A' if counter['A'] >= counter['N'] else 'N')
        
    return ''.join(ans)


def solution(survey, choices):
    score_counter = Counter()
    
    for s,c in zip (survey, choices):
        a,b = s[0],s[1]
        
        if c == 1:
            score_counter[a] += 3
        if c == 2:
            score_counter[a] += 2
        if c == 3:
            score_counter[a] += 1
        if c == 5:
            score_counter[b] += 1
        if c == 6:
            score_counter[b] += 2
        if c == 7:
            score_counter[b] += 3
    
    return score(score_counter)

```
