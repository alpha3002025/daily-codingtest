# python (my)
```python
from collections import Counter
def solution(participant, completion):
    counter = Counter(participant)
    
    for p in completion:
        if p in counter:
            counter[p]-=1
        else:
            return p
    
    for name, count in counter.items():
        if count >= 1:
            return name
    
    return ''
```


# python (책) : Counter on place
테이블을 따로 안만들고 그 자리에서 on place 형식으로 카운팅하는 방식
```python
from collections import Counter

def solution(participant, completion):
    complete_counter = Counter(completion)
    
    for p in participant:
        if p in complete_counter and complete_counter[p] > 0:
            complete_counter[p]-=1
        else:
            return p
```



# python (책): counter 의 차집합
```python
from collections import Counter
def solution(participant, completion):
    p_counter = Counter(participant)
    c_counter = Counter(completion)
    return ''.join(list(p_counter - c_counter))
```
