def solution(wallpaper):
    """
    왼쪽 상단
    - row : min
    - col : min
    
    오른쪽 하단
    - row : max
    - col : max
    """
    
    EMPTY = "."
    FILE = "#"
    
    min_r, min_c = (len(wallpaper), len(wallpaper[0]))
    max_r, max_c = (0,0)
    for r in range(len(wallpaper)):
        for c in range(len(wallpaper[r])):
            data = wallpaper[r][c]
            
            if data == FILE:
                min_r = min(min_r, r)
                min_c = min(min_c, c)
                max_r = max(max_r, r)
                max_c = max(max_c, c)
                
    
    return [min_r,min_c, max_r+1,max_c+1]
