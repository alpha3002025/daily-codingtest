```python
def solution(arr1, arr2):
    row_size = len(arr1)
    col_size = len(arr1[0])
    
    answer = []
    
    for i in range(row_size):
        row = []
        for j in range(col_size):
            row.append(arr1[i][j] + arr2[i][j])
        answer.append(row)
    return answer
```
