# kotlin
```kotlin
class Solution {
    fun solution(babbling: Array<String>): Int {
        val regex1 = "aya|ye|woo|ma".toRegex()
        
        return babbling
            .map{it.replace(regex1, "@")}
            .map{it.replace("@","")}
            .filter{it.length == 0}
            .size
    }
}
```


# python
```python
def solution(babbling):
    words = ["aya", "ye", "woo", "ma"]
    
    cnt = 0
    for b in babbling:
        for w in words:
            b = b.replace(w, ' ')
        
        b = b.strip()
        if b == "":
            cnt+=1
    
    return cnt
```
