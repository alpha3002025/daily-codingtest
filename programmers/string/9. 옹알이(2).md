```python
def solution(babbling):
    word = ["aya", "ye", "woo", "ma"]
    count = 0
    
    for b in babbling:
        for w in word:
            if w*2 in b:
                break
            b = b.replace(w, " ")
        
        if not b.strip():
            count+=1
    
    return count
```
