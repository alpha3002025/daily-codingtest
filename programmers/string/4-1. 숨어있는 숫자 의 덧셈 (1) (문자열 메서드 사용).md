# python (my)
```python
def solution(my_string):
    answer = 0
    
    for s in my_string:
        if s.isdigit():
            answer += int(s)
        
    return answer
```

def solution(my_string):
    only_numbers = ''.join(s if s.isdigit() else ' ' for s in my_string)
    return sum(int(s) for s in only_numbers.split())

# kotlin
```kotlin
class Solution {
    fun solution(my_string: String): Int {
        return my_string
            .filter{it.isDigit()}
            .map{it.toString()}
            .sumOf{it.toInt()}
    }
}
```
