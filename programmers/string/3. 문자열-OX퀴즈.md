# python
```python
def solution(quiz):
    answer = []
    
    for query in quiz:
        left, right = query.split("=")
        n1, op, n2 = left.split()
        
        curr = ""
        if op == '-':
            curr = int(n1) - int(n2)
        else:
            curr = int(n1) + int(n2)
        
        answer.append('O' if int(curr) == int(right) else 'X')
    
    return answer
```

class Solution {
    fun solution(quiz: Array<String>): Array<String> {
        var answer: Array<String> = arrayOf<String>()
        
        return quiz.map{
            val (left, op, right, equal, result) = it.split(" ")

        }.toTypedArray()
    }
}

```python
def solution(quiz):
    answer = []
    
    for q in quiz:
        left, right = q.split("=")
        
        elements = left.split()
        op = ''
        acc = 0
        
        for e in elements:
            if e in ['+','-']:
                op = e
            else:
                if op == '+':
                    acc += int(e)
                elif op == '-':
                    acc -= int(e)
                else:
                    acc += int(e)

        if acc == int(right):
            answer.append("O")
        else:
            answer.append("X")
    
    return answer
```
