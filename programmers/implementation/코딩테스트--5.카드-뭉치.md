```python
def solution(cards1, cards2, goal):
    c1, c2 = 0,0
    for s in goal:
        if c1 < len(cards1) and cards1[c1] == s:
            c1+=1
        elif c2 < len(cards2) and cards2[c2] == s:
            c2+=1
        else:
            return "No"
    return "Yes"
```
